35// This function will locate the path to our application on OS X,
36// unlike windows you can not rely on the curent working directory
37// for locating your configuration files and resources.
38std::string macBundlePath()
39{
40 char path[1024];
41 CFBundleRef mainBundle = CFBundleGetMainBundle();
42 assert(mainBundle);
43
44 CFURLRef mainBundleURL = CFBundleCopyBundleURL(mainBundle);
45 assert(mainBundleURL);
46
47 CFStringRef cfStringRef = CFURLCopyFileSystemPath( mainBundleURL, kCFURLPOSIXPathStyle);
48 assert(cfStringRef);
49
50 CFStringGetCString(cfStringRef, path, 1024, kCFStringEncodingASCII);
51
52 CFRelease(mainBundleURL);
53 CFRelease(cfStringRef);
54
55 return std::string(path);
56}

169 virtual bool configure(void)
170 {
171 // Show the configuration dialog and initialise the system
172 // You can skip this and use root.restoreConfig() to load configuration
173 // settings if you were sure there are valid ones saved in ogre.cfg
174 if(mRoot->showConfigDialog())
175 {
176 // If returned true, user clicked OK so initialise
177 // Here we choose to let the system create a default rendering window by passing 'true'
178 mWindow = mRoot->initialise(true);
179 mWindow->setDeactivateOnFocusChange(false);
180 return true;
