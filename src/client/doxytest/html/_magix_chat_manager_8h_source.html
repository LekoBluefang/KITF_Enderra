482 String tCaption = prefixChatLine(caption,sayer,type);

488 //size caption
489 const Font *pFont = dynamic_cast<Ogre::Font*>(Ogre::FontManager::getSingleton().getByName("Tahoma").getPointer());
490 const Real tHeight = charHeight;

492 int tSpacePos = -1;

500 tWidthFromSpace += pFont->getGlyphAspectRatio(0x0030);

505 tWidthFromSpace += pFont->getGlyphAspectRatio(tCaption[i]);
506 }
507
508 if(tCaption[i] == ' ' && i>tPrefix)
509 {
510 tSpacePos = i;
511 tWidthFromSpace = pFont->getGlyphAspectRatio(0x0030);
512 }
513 if(tTextWidth*tHeight>=boxWidth)
514 {
515 if(tSpacePos!=-1)
516 {
517 tCaption[tSpacePos] = '\n';
518 tTextWidth = tWidthFromSpace;
519 tSpacePos = -1;
520 }
521 else
522 {
523 tCaption.insert(i+1,String(1,'\n'));
524 tTextWidth = 0;
525 }
526 }
527 }
528 }
529
530 tCaption.erase(0,tPrefix);
531 tCaption.erase(tCaption.length()-tPostfix,tPostfix);
