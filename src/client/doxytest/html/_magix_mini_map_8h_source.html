272 bool isInMarker(OverlayElement *marker, const Real &x, const Real &y)
273 {
274 if(!mMap->contains(x,y))return false;
275 const Real tX = mMap->getLeft() + marker->getLeft();
276 const Real tY = mMap->getTop() + marker->getTop();
277 const Real tQuartWidth = marker->getWidth()/4;
278 const Real tQuartHeight = marker->getHeight()/4;
279
280 if(x >= tX+tQuartWidth && x <= tX+marker->getWidth()-tQuartWidth
281 && y >= tY+tQuartHeight && y <= tY+marker->getHeight()-tQuartHeight)
282 return true;
283 return false;
284 }
