198void MovableTextOverlay::_computeTextSize()
199{
200 //compute text width/height
201 const Font *pFont = mAttrs->getFont();
202
203 vector<String>::type tText = StringUtil::split(mCaption,"\n");
204 Real tLongestWidth = 0;
205
206 for(int i=0; i<int(tText.size()); i++)
207 {
208 String tLine = tText[i];
209 Real tTextWidth = 0;
210 for(Ogre::String::iterator i = tLine.begin(); i < tLine.end();i++)
211 {
212 if(*i == 0x0020)
213 tTextWidth += pFont->getGlyphAspectRatio(0x0030);
214 else
215 {
216 tTextWidth += pFont->getGlyphAspectRatio(*i);
217 }
218 }
219 if(tTextWidth>tLongestWidth)tLongestWidth = tTextWidth;
220 }
221
222 mTextWidth = tLongestWidth;
223 mTextWidth *= mAttrs->getCharacterHeight();
224 mNoOfLines = int(tText.size());
225}

227void MovableTextOverlay::_getMinMaxEdgesOfAABBIn2D(Ogre::Real& MinX, Ogre::Real& MinY, Ogre::Real& MaxX, Ogre::Real& MaxY, bool getTop)
228{
229 const Ogre::Camera* mpCam = mAttrs->getCamera();
230
231 MinX = 0;
232 MinY = 0;
233 MaxX = 0;
234 MaxY = 0;
235
236 Ogre::Real X[4];// the 2D dots of the AABB in screencoordinates
237 Ogre::Real Y[4];
238
239 if(!mpMov->isInScene())
240 return;
241
242 const Ogre::AxisAlignedBox &AABB = mpMov->getWorldBoundingBox(true);// the AABB of the target
243 const Ogre::Vector3 CornersOfTopAABB[4] =
244 {getTop ? AABB.getCorner(AxisAlignedBox::FAR_LEFT_TOP) : AABB.getCorner(AxisAlignedBox::FAR_LEFT_BOTTOM),
245 getTop ? AABB.getCorner(AxisAlignedBox::FAR_RIGHT_TOP) : AABB.getCorner(AxisAlignedBox::FAR_RIGHT_BOTTOM),
246 getTop ? AABB.getCorner(AxisAlignedBox::NEAR_LEFT_TOP) : AABB.getCorner(AxisAlignedBox::NEAR_LEFT_BOTTOM),
247 getTop ? AABB.getCorner(AxisAlignedBox::NEAR_RIGHT_TOP) : AABB.getCorner(AxisAlignedBox::NEAR_RIGHT_BOTTOM)};
248
249 Ogre::Vector3 CameraPlainNormal = mpCam->getDerivedOrientation().zAxis();//The normal vector of the plaine.this points directly infront of the cam
250
251 Ogre::Plane CameraPlain = Plane(CameraPlainNormal,mpCam->getDerivedPosition());//the plaine that devides the space bevor and behin the cam
252
253 for (int i = 0; i < 4; i++)
254 {
255 X[i] = 0;
256 Y[i] = 0;
257
258 _getScreenCoordinates(CornersOfTopAABB[i],X[i],Y[i]);// transfor into 2d dots
259
260 if(CameraPlain.getSide(CornersOfTopAABB[i]) == Plane::NEGATIVE_SIDE)
261 {
262
263 if(i == 0)// accept the first set of values, no matter how bad it might be.
264 {
265 MinX = X[i];
266 MinY = Y[i];
267 MaxX = X[i];
268 MaxY = Y[i];
269 }
270 else// now compare if you get "better" values
271 {
272 if(MinX > X[i])// get the x minimum
273 {
274 MinX = X[i];
275 }
276
277 if(MinY > Y[i])// get the y minimum
278 {
279 MinY = Y[i];
280 }
281
282 if(MaxX < X[i])// get the x maximum
283 {
284 MaxX = X[i];
285 }
286
287 if(MaxY < Y[i])// get the y maximum
288 {
289 MaxY = Y[i];
290 }
291 }
292 }
293 else
294 {
295 MinX = 0;
296 MinY = 0;
297 MaxX = 0;
298 MaxY = 0;
299 break;
300 }
301 }

304void MovableTextOverlay::_getScreenCoordinates(const Ogre::Vector3& position, Ogre::Real& x, Ogre::Real& y)
305{
306 const Ogre::Camera* mpCam = mAttrs->getCamera();
307 Vector3 hcsPosition = mpCam->getProjectionMatrix() * (mpCam->getViewMatrix() * position);
308
309 x = 1.0f - ((hcsPosition.x * 0.5f) + 0.5f);// 0 <= x <= 1 // left := 0,right := 1
310 y = ((hcsPosition.y * 0.5f) + 0.5f);// 0 <= y <= 1 // bottom := 0,top := 1
311}

331void MovableTextOverlay::update(const Real &timeSincelastFrame, bool placeAtTop, const Real &borderSizeOffset, bool chaseUpdate)
332{
333 static Real timeTillUpdate = 0;
334
335 timeTillUpdate -= timeSincelastFrame;
336 if(timeTillUpdate > 0)
337 {
338 return;
339 }
340
341 timeTillUpdate = mUpdateFrequency;
342
343 Ogre::Real min_x, max_x, min_y, max_y;
344 _getMinMaxEdgesOfAABBIn2D(min_x, min_y, max_x, max_y, placeAtTop);
345
346 if(mNeedUpdate)
347 {
348 _computeTextSize();
349 mNeedUpdate = false;
350 }
351
352 Real relTextWidth = mTextWidth / OverlayManager::getSingleton().getViewportWidth();
353 Real relTextHeight = mAttrs->getCharacterHeight()*mNoOfLines / OverlayManager::getSingleton().getViewportHeight();
354
355 Real tX = 1-(min_x + max_x + relTextWidth)/2;
356 Real tY = 1-max_y-(placeAtTop?relTextHeight+borderSizeOffset:0) + mYOffset;
357
358 //container "chases" towards update position
359 if(chaseUpdate)
360 {
361 tX = (tX - mpOvContainer->getLeft()) * (Real)0.08;
362 tY = (tY - mpOvContainer->getTop()) * (Real)0.08;
363
364 mpOvContainer->setPosition(mpOvContainer->getLeft()+tX, mpOvContainer->getTop()+tY);
365 }
366 else
367 {
368 mpOvContainer->setPosition(tX, tY);
369 }
370
371 mpOvContainer->setDimensions(relTextWidth+borderSizeOffset, relTextHeight+borderSizeOffset);
372 if(borderSizeOffset!=0)
373 {
374 mpOvText->setPosition(0,borderSizeOffset*600);
375 }
376
377 if((min_x+relTextWidth>0.0) && (max_x-relTextWidth<1.0) && (min_y+relTextHeight>0.0) && (max_y-relTextHeight<1.0)
378 && !(min_x==0&&max_x==0&&min_y==0&&max_y==0) )
379 {
380 mOnScreen = true;
381 }
382 else
383 {
384 mOnScreen = false;
385 }
