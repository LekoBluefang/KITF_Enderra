150int fontCacheInit(Ogre::String fontName)
151{
152 Ogre::FontPtr font_org = Ogre::FontManager::getSingleton().getByName(fontName);
153 Ogre::FontPtr font_cache = Ogre::FontManager::getSingleton().getByName(fontName + cache_name);
154
155 // base font not even existing?
156 if (font_org.isNull()) return 1;
157 // using texture fonts is a bit stupid ...
158 if (font_org->getType() != Ogre::FT_TRUETYPE) return 1;
159
160 int res = 0;
161 if (font_cache.isNull())
162 res = generateFontTexture(fontName);
163
164 if (font_org->isLoaded()) font_org->unload();
